50 static void pp_mil_k2(fp2_t r, ep_t t, ep_t p, ep_t q, bn_t a) {
51  fp2_t l;
52  ep_t _q;
53 
54  fp2_null(l);
55  ep_null(_q);
56 
57  TRY {
58  fp2_new(l);
59  ep_new(_q);
60 
61  fp2_zero(l);
62  fp2_zero(r);
63  fp_set_dig(r[0], 1);
64  ep_copy(t, p);
65 
66  ep_neg(_q, q);
67 
68  for (int i = bn_bits(a) - 2; i >= 0; i--) {
69  fp2_sqr(r, r);
70  pp_dbl_k2(l, t, t, _q);
71  fp2_mul(r, r, l);
72  if (bn_get_bit(a, i)) {
73  pp_add_k2(l, t, p, q);
74  fp2_mul(r, r, l);
75  }
76  }
77  }
78  CATCH_ANY {
79  THROW(ERR_CAUGHT);
80  }
81  FINALLY {
82  fp2_free(l);
83  ep_free(_q);
84  }
85 }

108  fp12_zero(l);
109  fp12_zero(r);
110  fp_set_dig(r[0][0][0], 1);

113  /* Precomputing. */

125  pp_add_k12(l, t, q, p);

465 void pp_map_oatep_k12(fp12_t r, ep_t p, ep2_t q) {
466  ep2_t t;
467  bn_t a;
468  int len = FP_BITS, s[FP_BITS];
469 
470  ep2_null(t);
471  bn_null(a);
472 
473  TRY {
474  ep2_new(t);
475  bn_new(a);
476 
477  fp_param_get_var(a);
478  bn_mul_dig(a, a, 6);
479  bn_add_dig(a, a, 2);
480  fp_param_get_map(s, &len);
481 
482  switch (ep_param_get()) {
483  case BN_P158:
484  case BN_P254:
485  case BN_P256:
486  case BN_P638:
487  /* r = f_{|a|,Q}(P). */
488  pp_mil_sps_k12(r, t, q, p, s, len);
489  if (bn_sign(a) == BN_NEG) {
490  /* f_{-a,Q}(P) = 1/f_{a,Q}(P). */
491  fp12_inv_uni(r, r);
492  ep2_neg(t, t);
493  }
494  pp_fin_k12_oatep(r, t, q, p);
495  pp_exp_k12(r, r);
496  break;
497  case B12_P638:
498  /* r = f_{|a|,Q}(P). */
499  pp_mil_sps_k12(r, t, q, p, s, len);
500  if (bn_sign(a) == BN_NEG) {
501  fp12_inv_uni(r, r);
502  ep2_neg(t, t);
503  }
504  pp_exp_k12(r, r);
505  break;
506  }
507  }
508  CATCH_ANY {
509  THROW(ERR_CAUGHT);
510  }
511  FINALLY {
512  ep2_free(t);
513  bn_free(a);
514  }
515 }
