47 int cp_sokaka_gen(bn_t s) {
48  bn_t n;
49  int result = STS_OK;
50 
51  bn_null(n);
52 
53  TRY {
54  bn_new(n);
55 
56  g1_get_ord(n);
57 
58  do {
59  bn_rand(s, BN_POS, bn_bits(n));

62  }

67  bn_free(n);
