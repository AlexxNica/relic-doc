170 void ep2_curve_init(void) {
171  ctx_t *ctx = core_get();
172 
173 #ifdef EP_PRECO
174  for (int i = 0; i < EP_TABLE; i++) {
175  ctx->ep2_ptr[i] = &(ctx->ep2_pre[i]);
176  }
177 #endif
178 
179 #if ALLOC == STATIC || ALLOC == DYNAMIC || ALLOC == STACK
180  ctx->ep2_g.x[0] = ctx->ep2_gx[0];
181  ctx->ep2_g.x[1] = ctx->ep2_gx[1];
182  ctx->ep2_g.y[0] = ctx->ep2_gy[0];
183  ctx->ep2_g.y[1] = ctx->ep2_gy[1];
184  ctx->ep2_g.z[0] = ctx->ep2_gz[0];
185  ctx->ep2_g.z[1] = ctx->ep2_gz[1];
186 #endif
187 
188 #ifdef EP_PRECO
189 #if ALLOC == STATIC || ALLOC == DYNAMIC
190  for (int i = 0; i < EP_TABLE; i++) {
191  fp2_new(ctx->ep2_pre[i].x);
192  fp2_new(ctx->ep2_pre[i].y);
193  fp2_new(ctx->ep2_pre[i].z);
194  }
195 #elif ALLOC == STACK
196  for (int i = 0; i < EP_TABLE; i++) {
197  ctx->ep2_pre[i].x[0] = ctx->_ep2_pre[3 * i][0];
198  ctx->ep2_pre[i].x[1] = ctx->_ep2_pre[3 * i][1];
199  ctx->ep2_pre[i].y[0] = ctx->_ep2_pre[3 * i + 1][0];
200  ctx->ep2_pre[i].y[1] = ctx->_ep2_pre[3 * i + 1][1];
201  ctx->ep2_pre[i].z[0] = ctx->_ep2_pre[3 * i + 2][0];
202  ctx->ep2_pre[i].z[1] = ctx->_ep2_pre[3 * i + 2][1];
203  }
204 #endif
205 #endif
206  ep2_set_infty(&(ctx->ep2_g));
207  bn_init(&(ctx->ep2_r), FP_DIGS);
208  bn_init(&(ctx->ep2_h), FP_DIGS);
209 }

269 void ep2_curve_set_twist(int type) {
270  char str[2 * FP_BYTES + 1];
271  ctx_t *ctx = core_get();
272  ep2_t g;
273  fp2_t a;
274  fp2_t b;
275  bn_t r;
276 
277  ep2_null(g);
278  fp2_null(a);
279  fp2_null(b);
280  bn_null(r);
281 
282  ctx->ep2_is_twist = 0;
283  if (type == EP_MTYPE || type == EP_DTYPE) {
284  ctx->ep2_is_twist = type;
285  } else {
286  return;
287  }
288 
289  TRY {
290  ep2_new(g);
291  fp2_new(a);
292  fp2_new(b);
293  bn_new(r);
294 
295  switch (ep_param_get()) {
296 #if FP_PRIME == 80
297  case GLS_P160:
298  ASSIGN(GLS_P160);
299  break;
300 #elif FP_PRIME == 158
301  case BN_P158:
302  ASSIGN(BN_P158);
303  break;
304 #elif FP_PRIME == 254
305  case BN_P254:
306  ASSIGN(BN_P254);
307  break;
308 #elif FP_PRIME == 256
309  case BN_P256:
310  ASSIGN(BN_P256);
311  break;
312 #elif FP_PRIME == 638
313  case BN_P638:
314  ASSIGN(BN_P638);
315  break;
316  case B12_P638:
317  ASSIGN(B12_P638);
318  break;
319 #endif
320  default:
321  (void)str;
322  THROW(ERR_NO_VALID);
323  break;
324  }
325 
326  fp2_zero(g->z);
327  fp_set_dig(g->z[0], 1);
328  g->norm = 1;
329 
330  ep2_copy(&(ctx->ep2_g), g);
331  fp_copy(ctx->ep2_a[0], a[0]);
332  fp_copy(ctx->ep2_a[1], a[1]);
333  fp_copy(ctx->ep2_b[0], b[0]);
334  fp_copy(ctx->ep2_b[1], b[1]);
335  bn_copy(&(ctx->ep2_r), r);
336  bn_set_dig(&(ctx->ep2_h), 1);
337 
338  /* I don't have a better place for this. */
339  fp_prime_calc();
340 
341 #if defined(EP_PRECO)
342  ep2_mul_pre((ep2_t *)ep2_curve_get_tab(), &(ctx->ep2_g));
343 #endif
344  }
345  CATCH_ANY {
346  THROW(ERR_CAUGHT);
347  }
348  FINALLY {
349  ep2_free(g);
350  fp2_free(a);
351  fp2_free(b);
352  bn_free(r);
353  }
354 }
