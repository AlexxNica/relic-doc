39 void fb2_slv2(fb2_t c, fb2_t a) {
40  /* Compute c_0 = a_0 + a_1. */
41  fb_add(c[0], a[0], a[1]);
42  /* Compute c_1^2 + c_1 = a_1. */
43  fb_slv(c[1], a[1]);
44  /* Compute c_0 = a_0 + a_1 + c_1 + Tr(c_1). */
45  fb_add(c[0], c[0], c[1]);
46  fb_add_dig(c[0], c[0], fb_trc(c[1]));
47  /* Make Tr(c_0) = 0. */
48  fb_slv(c[0], c[0]);
49  /* Compute c_0^2 + c_0 = c_0. */
50  fb_add_dig(c[1], c[1], fb_trc(c[1]));
51 }
