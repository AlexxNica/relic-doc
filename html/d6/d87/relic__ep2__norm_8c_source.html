47 static void ep2_norm_imp(ep2_t r, ep2_t p) {
48  if (!p->norm) {
49  fp2_t t0, t1;
50 
51  fp2_null(t0);
52  fp2_null(t1);
53 
54  TRY {
55  fp2_new(t0);
56  fp2_new(t1);
57 
58  fp2_inv(t1, p->z);
59  fp2_sqr(t0, t1);
60  fp2_mul(r->x, p->x, t0);
61  fp2_mul(t0, t0, t1);
62  fp2_mul(r->y, p->y, t0);
63  fp_zero(r->z[0]);
64  fp_zero(r->z[1]);
65  fp_set_dig(r->z[0], 1);
66  }
67  CATCH_ANY {
68  THROW(ERR_CAUGHT);
69  }
70  FINALLY {
71  fp2_free(t0);
72  fp2_free(t1);
73  }
74  }
75 
76  r->norm = 1;
77 }
