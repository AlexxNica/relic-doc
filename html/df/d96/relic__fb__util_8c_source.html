202 void fb_read_str(fb_t a, const char *str, int len, int radix) {
203  int i, j, l;
204  char c;
205  dig_t carry;
206 
207  fb_zero(a);
208 
209  l = log_radix(radix);
210  if (!valid_radix(radix)) {
211  THROW(ERR_NO_VALID);
212  }
213 
214  j = 0;
215  while (str[j] && j < len) {
216  c = (char)((radix < 36) ? TOUPPER(str[j]) : str[j]);
217  for (i = 0; i < 64; i++) {
218  if (c == util_conv_char(i)) {
219  break;
220  }
221  }
222 
223  if (i < radix) {
224  carry = fb_lshb_low(a, a, l);
225  if (carry != 0) {
226  THROW(ERR_NO_BUFFER);
227  }
228  fb_add_dig(a, a, (dig_t)i);
229  } else {
230  break;
231  }
232  j++;
233  }

236 void fb_write_str(char *str, int len, const fb_t a, int radix) {
237  fb_t t;
238  int d, l, i, j;
239  char c;
240 
241  fb_null(t);
242 
243  l = fb_size_str(a, radix);
244  if (len < l) {
245  THROW(ERR_NO_BUFFER);
246  }
247  len = l;
248 
249  l = log_radix(radix);
250  if (!valid_radix(radix)) {
251  THROW(ERR_NO_VALID)
252  }
253 
254  if (fb_is_zero(a) == 1) {
255  *str++ = '0';
256  *str = '\0';
257  return;
258  }
259 
260  TRY {
261  fb_new(t);
262  fb_copy(t, a);
263 
264  j = 0;
265  while (!fb_is_zero(t)) {
266  d = t[0] % radix;
267  fb_rshb_low(t, t, l);
268  str[j] = util_conv_char(d);
269  j++;
270  }
271 
272  /* Reverse the digits of the string. */
273  i = 0;
274  j = len - 2;
275  while (i < j) {
276  c = str[i];
277  str[i] = str[j];
278  str[j] = c;
279  ++i;
280  --j;
281  }
282 
283  str[len - 1] = '\0';
284  }
285  CATCH_ANY {
286  THROW(ERR_CAUGHT);
287  }
288  FINALLY {
289  fb_free(t);
290  }
291 }
