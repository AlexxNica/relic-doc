75 void fp2_read_bin(fp2_t a, uint8_t *bin, int len) {
76  if (len != FP_BYTES + 1 && len != 2 * FP_BYTES) {
77  THROW(ERR_NO_BUFFER);
78  }
79  if (len == FP_BYTES + 1) {

90 void fp2_write_bin(uint8_t *bin, int len, fp2_t a, int pack) {
91  fp2_t t;
92 
93  fp2_null(t);
94 
95  TRY {
96  fp2_new(t);
97 
98  if (pack && fp2_test_uni(a)) {
99  if (len != FP_BYTES + 1) {
100  THROW(ERR_NO_BUFFER);
101  } else {
102  fp2_pck(t, t);
103  fp_write_bin(bin, FP_BYTES, a[0]);
104  bin[FP_BYTES] = fp_get_bit(a[1], 0);
105  }
106  } else {
107  if (len != 2 * FP_BYTES) {
108  THROW(ERR_NO_BUFFER);
109  } else {
110  fp_write_bin(bin, FP_BYTES, a[0]);
111  fp_write_bin(bin + FP_BYTES, FP_BYTES, a[1]);
112  }
113  }
114  } CATCH_ANY {
115  THROW(ERR_CAUGHT);
116  }
117  FINALLY {
118  fp2_free(t);
119  }
120 }

263 void fp12_read_bin(fp12_t a, uint8_t *bin, int len) {
264  if (len != 8 * FP_BYTES && len != 12 * FP_BYTES) {
265  THROW(ERR_NO_BUFFER);
266  }
267  if (len == 8 * FP_BYTES) {
268  fp2_zero(a[0][0]);
269  fp2_read_bin(a[0][1], bin, 2 * FP_BYTES);
270  fp2_read_bin(a[0][2], bin + 2 * FP_BYTES, 2 * FP_BYTES);
271  fp2_read_bin(a[1][0], bin + 4 * FP_BYTES, 2 * FP_BYTES);
272  fp2_zero(a[1][1]);
273  fp2_read_bin(a[1][2], bin + 6 * FP_BYTES, 2 * FP_BYTES);
274  fp12_back_cyc(a, a);
275  }
276  if (len == 12 * FP_BYTES) {
277  fp6_read_bin(a[0], bin, 6 * FP_BYTES);
278  fp6_read_bin(a[1], bin + 6 * FP_BYTES, 6 * FP_BYTES);
