48 /*
49  * Computes the hash derivation function.
50  *
51  * param[out] out - the result.
52  * param[in] out_len - the number of bytes to return.
53  * param[in] in - the input string.
54  * param[in] in_len - the number of bytes in the input.
55  */
56 static void rand_hash(uint8_t *out, int out_len, uint8_t *in,
57  int in_len) {
58  uint32_t j = util_conv_big(8 * out_len);
59  int len = CEIL(out_len, MD_LEN);
60  uint8_t buf[1 + sizeof(uint32_t) + in_len], hash[MD_LEN];
61 
62  buf[0] = 1;
63  memcpy(buf + 1, &j, sizeof(uint32_t));
64  memcpy(buf + 1 + sizeof(uint32_t), in, in_len);
65 
66  for (int i = 0; i < len; i++) {
67  /* h = Hash(counter || bits_to_return || input_string) */
68  md_map(hash, buf, 1 + sizeof(uint32_t) + in_len);
69  /* temp = temp || h */
70  memcpy(out, hash, MIN(MD_LEN, out_len));
71  out += MD_LEN;
72  out_len -= MD_LEN;
73  /* counter = counter + 1 */
74  buf[0]++;
75  }
76 }

84 static int rand_inc(uint8_t *data, int size, int digit) {
85  int carry = digit;
86  for (int i = size - 1; i >= 0; i--) {
87  int16_t s;
88  s = (data[i] + carry);
89  data[i] = s & 0xFF;
90  carry = s >> 8;
91  }
92  return carry;
93 }

101 static int rand_add(uint8_t *state, uint8_t *hash, int size) {
102  int carry = 0;
103  for (int i = size - 1; i >= 0; i--) {
104  /* Make sure carries are detected. */
105  int16_t s;
106  s = (state[i] + hash[i] + carry);
107  state[i] = s & 0xFF;
108  carry = s >> 8;
109  }
110  return carry;
111 }

119 static void rand_gen(uint8_t *out, int out_len) {
120  int m = CEIL(out_len, MD_LEN);
121  uint8_t hash[MD_LEN], data[(RAND_SIZE - 1)/2];
122  ctx_t *ctx = core_get();
123 
124  /* data = V */
125  memcpy(data, ctx->rand + 1, (RAND_SIZE - 1)/2);
126  for (int i = 0; i < m; i++) {
127  /* w_i = Hash(data) */
128  md_map(hash, data, sizeof(data));
129  /* W = W || w_i */
130  memcpy(out, hash, MIN(MD_LEN, out_len));
131  out += MD_LEN;
132  out_len -= MD_LEN;
133  /* data = data + 1 mod 2^b. */
134  rand_inc(data, (RAND_SIZE - 1)/2, 1);
135  }
136 }

146 void rand_bytes(uint8_t *buf, int size) {
147  uint8_t hash[MD_LEN];
148  int carry, len = (RAND_SIZE - 1)/2;
149  ctx_t *ctx = core_get();
150 
151  if (sizeof(int) > 2 && size > (1 << 16)) {
152  THROW(ERR_NO_VALID);
153  }
154 
155  /* buf = hash_gen(size) */
156  rand_gen(buf, size);
157  /* H = hash(03 || V) */
158  ctx->rand[0] = 0x3;
159  md_map(hash, ctx->rand, 1 + len);
160  /* V = V + H + C + reseed_counter. */
161  rand_add(ctx->rand + 1, ctx->rand + 1 + len, len);
162  carry = rand_add(ctx->rand + 1 + (len - MD_LEN), hash, MD_LEN);
163  rand_inc(ctx->rand, len - MD_LEN + 1, carry);
164  rand_inc(ctx->rand, len + 1, ctx->counter);
165  ctx->counter = ctx->counter + 1;
166 }

168 void rand_seed(uint8_t *buf, int size) {
169  ctx_t *ctx = core_get();
170  int len = (RAND_SIZE - 1) / 2;
171 
172  if (size <= 0) {
173  THROW(ERR_NO_VALID);
174  }
175 
176  if (sizeof(int) > 4 && size > (1 << 32)) {
177  THROW(ERR_NO_VALID);
178  }
179 
180  ctx->rand[0] = 0x0;
181  if (ctx->seeded == 0) {
182  /* V = hash_df(seed). */
183  rand_hash(ctx->rand + 1, len, buf, size);
184  /* C = hash_df(00 || V). */
185  rand_hash(ctx->rand + 1 + len, len, ctx->rand, len + 1);
186  } else {

188  uint8_t tmp[1 + len + size];

193  /* C = hash_df(00 || V). */
194  rand_hash(ctx->rand + 1 + len, len, ctx->rand, len + 1);
195  }
196  ctx->counter = ctx->seeded = 1;
197 }
