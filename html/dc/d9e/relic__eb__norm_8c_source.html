50 static void eb_norm_imp(eb_t r, const eb_t p, int flag) {
51  if (!p->norm) {
52  if (flag) {
53  fb_copy(r->z, p->z);
54  } else {
55  fb_inv(r->z, p->z);
56  }
57  fb_mul(r->x, p->x, r->z);
58  fb_sqr(r->z, r->z);
59  fb_mul(r->y, p->y, r->z);
60  fb_set_dig(r->z, 1);
61  }
62 
63  r->norm = 1;
64 }

85 void eb_norm(eb_t r, const eb_t p) {
86  if (eb_is_infty(p)) {
87  eb_set_infty(r);
88  return;
89  }
90 
91  if (p->norm == 1) {
92  /* If the point is represented in affine coordinates, we just copy it. */
93  eb_copy(r, p);
94  return;
95  }
96 
97  if (p->norm == 2) {
98  eb_norm_hlv(r, p);
99  return;
100  }
101 
102 #if EB_ADD == PROJC || !defined(STRIP)
103  eb_norm_imp(r, p, 0);
104 #endif /* EB_ADD == PROJC */
105 }

107 void eb_norm_sim(eb_t *r, const eb_t *t, int n) {
108  int i;
109  fb_t a[n];
110 
111  if (n == 1) {
112  eb_norm(r[0], t[0]);
113  return;
114  }
115 
116  for (i = 0; i < n; i++) {
117  fb_null(a[i]);
118  }
119 
120  TRY {
121  for (i = 0; i < n; i++) {
122  fb_new(a[i]);
123  if (!eb_is_infty(t[i])) {
124  fb_copy(a[i], t[i]->z);
125  } else {
126  fb_set_dig(a[i], 1);
127  }
128  }
129 
130  fb_inv_sim(a, (const fb_t *)a, n);
131 
132  for (i = 0; i < n; i++) {
133  fb_copy(r[i]->x, t[i]->x);
134  fb_copy(r[i]->y, t[i]->y);
135  if (!eb_is_infty(t[i])) {
136  fb_copy(r[i]->z, a[i]);
137  }
138  }
139  }
140  CATCH_ANY {
141  THROW(ERR_CAUGHT);
142  }
143  FINALLY {
144  for (i = 0; i < n; i++) {
145  fb_free(a[i]);
146  }
147  }
148 
149  for (i = 0; i < n; i++) {
150  eb_norm_imp(r[i], r[i], 1);
151  }
152 }
