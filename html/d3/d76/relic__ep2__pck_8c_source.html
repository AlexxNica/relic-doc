39 void ep2_pck(ep2_t r, ep2_t p) {

50 int ep2_upk(ep2_t r, ep2_t p) {
51  fp2_t t;
52  int result = 0;
53 
54  fp2_null(t);
55 
56  TRY {
57  fp2_new(t);
58 
59  ep2_rhs(t, p);
60 
61  /* t0 = sqrt(x1^3 + a * x1 + b). */
62  result = fp2_srt(t, t);
63 
64  if (result) {
65  /* Verify if least significant bit of the result matches the
66  * compressed y-coordinate. */
67  if (fp_get_bit(t[0], 0) != fp_get_bit(p->y[0], 0)) {
68  fp2_neg(t, t);
69  }
70  fp2_copy(r->x, p->x);
71  fp2_copy(r->y, t);
72  fp_set_dig(r->z[0], 1);
73  fp_zero(r->z[1]);
74  r->norm = 1;
75  }
76  }
77  CATCH_ANY {
78  THROW(ERR_CAUGHT);
79  }
80  FINALLY {
81  fp2_free(t);
82  }
83  return result;
84 }
