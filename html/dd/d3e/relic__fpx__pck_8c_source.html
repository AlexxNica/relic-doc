49 int fp2_upk(fp2_t c, fp2_t a) {
50  int result, b = fp_get_bit(a[1], 0);
51  fp_t t;
52 
53  fp_null(t);
54 
55  TRY {
56  fp_new(t);
57 
58  /* a_0^2 + a_1^2 = 1, thus a_1^2 = 1 - a_0^2. */
59  fp_sqr(t, a[0]);
60  fp_sub_dig(t, t, 1);
61  fp_neg(t, t);
62 
63  /* a1 = sqrt(a_0^2). */
64  result = fp_srt(t, t);
65 
66  if (result) {
67  /* Verify if least significant bit of the result matches the
68  * compressed second coordinate. */
69  if (fp_get_bit(t, 0) != b) {
70  fp_neg(t, t);
71  }
72  fp_copy(c[0], a[0]);
73  fp_copy(c[1], t);
74  }
75  } CATCH_ANY {
76  THROW(ERR_CAUGHT);
77  } FINALLY {
78  fp_free(t);
79  }
80 
81  return result;
82 }
