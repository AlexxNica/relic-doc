96 void ep_rhs(fp_t rhs, const ep_t p) {
97  fp_t t0;
98  fp_t t1;
99 
100  fp_null(t0);
101  fp_null(t1);
102 
103  TRY {
104  fp_new(t0);
105  fp_new(t1);
106 
107  /* t0 = x1^2. */
108  fp_sqr(t0, p->x);
109  /* t1 = x1^3. */
110  fp_mul(t1, t0, p->x);
111 
112  /* t1 = x1^3 + a * x1 + b. */
113  switch (ep_curve_opt_a()) {
114  case OPT_ZERO:
115  break;
116  case OPT_ONE:
117  fp_add(t1, t1, p->x);
118  break;
119 #if FP_RDC != MONTY
120  case OPT_DIGIT:
121  fp_mul_dig(t0, p->x, ep_curve_get_a()[0]);
122  fp_add(t1, t1, t0);
123  break;
124 #endif
125  default:
126  fp_mul(t0, p->x, ep_curve_get_a());
127  fp_add(t1, t1, t0);
128  break;
129  }
130 
131  switch (ep_curve_opt_b()) {
132  case OPT_ZERO:
133  break;
134  case OPT_ONE:
135  fp_add_dig(t1, t1, 1);
136  break;
137 #if FP_RDC != MONTY
138  case OPT_DIGIT:
139  fp_add_dig(t1, t1, ep_curve_get_b()[0]);
140  break;
141 #endif
142  default:
143  fp_add(t1, t1, ep_curve_get_b());
144  break;
145  }
146 
147  fp_copy(rhs, t1);
148 
149  } CATCH_ANY {
150  THROW(ERR_CAUGHT);
151  } FINALLY {
152  fp_free(t0);
153  fp_free(t1);
154  }
155 }

157 int ep_is_valid(const ep_t p) {
158  ep_t t;
159  int r = 0;
160 
161  ep_null(t);
162 
163  TRY {
164  ep_new(t);
165 
166  ep_norm(t, p);
167 
168  ep_rhs(t->x, t);
169  fp_sqr(t->y, t->y);
170  r = (fp_cmp(t->x, t->y) == CMP_EQ) || ep_is_infty(p);
171  } CATCH_ANY {
172  THROW(ERR_CAUGHT);
173  } FINALLY {
174  ep_free(t);
175  }
176  return r;

179 void ep_tab(ep_t *t, const ep_t p, int w) {
180  if (w > 2) {
181  ep_dbl(t[0], p);
182 #if defined(EP_MIXED)
183  ep_norm(t[0], t[0]);
184 #endif
185  ep_add(t[1], t[0], p);
186  for (int i = 2; i < (1 << (w - 2)); i++) {
187  ep_add(t[i], t[i - 1], t[0]);
188  }
189 #if defined(EP_MIXED)
190  ep_norm_sim(t + 1, (const ep_t *)t + 1, (1 << (w - 2)) - 1);
191 #endif
192  }
193  ep_copy(t[0], p);
194 }

230 void ep_read_bin(ep_t a, const uint8_t *bin, int len) {
231  if (len == 1) {
232  if (bin[0] == 0) {
233  ep_set_infty(a);
234  return;
235  } else {
236  THROW(ERR_NO_BUFFER);
237  return;
238  }
239  }
240 
241  if (len != (FP_BYTES + 1) && len != (2 * FP_BYTES + 1)) {
242  THROW(ERR_NO_BUFFER);
243  return;
244  }
245 
246  a->norm = 1;
247  fp_set_dig(a->z, 1);
248  fp_read_bin(a->x, bin + 1, FP_BYTES);
249  if (len == FP_BYTES + 1) {
250  switch(bin[0]) {
251  case 2:
252  fp_zero(a->y);
253  break;
254  case 3:
255  fp_zero(a->y);
256  fp_set_bit(a->y, 0, 1);
257  break;
258  default:
259  THROW(ERR_NO_VALID);
260  break;
261  }
262  ep_upk(a, a);
263  }
264 
265  if (len == 2 * FP_BYTES + 1) {
266  if (bin[0] == 4) {
267  fp_read_bin(a->y, bin + FP_BYTES + 1, FP_BYTES);
268  } else {
269  THROW(ERR_NO_VALID);
270  }
271  }
272 }

274 void ep_write_bin(uint8_t *bin, int len, const ep_t a, int pack) {
275  ep_t t;
276 
277  ep_null(t);
278 
279  if (ep_is_infty(a)) {
280  if (len != 1) {
281  THROW(ERR_NO_BUFFER);

283  bin[0] = 0;
284  return;
285  }
286  }
287 
288  TRY {

292 
293  if (pack) {

295  THROW(ERR_NO_BUFFER);
296  } else {
297  ep_pck(t, t);
298  bin[0] = 2 | fp_get_bit(t->y, 0);
299  fp_write_bin(bin + 1, FP_BYTES, t->x);
300  }

303  THROW(ERR_NO_BUFFER);
304  } else {
