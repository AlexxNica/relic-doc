47 int ep_upk(ep_t r, const ep_t p) {
48  fp_t t;
49  int result = 0;
50 
51  fp_null(t);
52 
53  TRY {
54  fp_new(t);
55 
56  ep_rhs(t, p);
57 
58  /* t0 = sqrt(x1^3 + a * x1 + b). */
59  result = fp_srt(t, t);
60 
61  if (result) {
62  /* Verify if least significant bit of the result matches the
63  * compressed y-coordinate. */
64  if (fp_get_bit(t, 0) != fp_get_bit(p->y, 0)) {
65  fp_neg(t, t);
66  }
67  fp_copy(r->x, p->x);
68  fp_copy(r->y, t);
69  fp_set_dig(r->z, 1);
70  r->norm = 1;
71  }
72  }
73  CATCH_ANY {
74  THROW(ERR_CAUGHT);
75  }
76  FINALLY {
77  fp_free(t);
78  }
79  return result;
