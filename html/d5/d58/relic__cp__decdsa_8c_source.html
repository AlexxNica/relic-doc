46 void cp_decdsa_gen(bn_t d_1, bn_t d_2, ec_t q) {
47  bn_t d, n, t;
48 
49  bn_null(d);
50  bn_null(n);
51  bn_null(t);
52 
53  TRY {
54  bn_new(d);
55  bn_new(n);
56  bn_new(t);
57 
58  ec_curve_get_ord(n);
59 
60  do {
61  bn_rand(d, BN_POS, bn_bits(n));
62  bn_mod(d, d, n);
63  } while (bn_is_zero(d));
64 
65  ec_mul_gen(q, d);
66 
67  /* Split private key d multiplicatively, by generating a random d_1
68  * and computing d_2 = d/d_1 (mod n) . */
69  do {
70  bn_rand(d_1, BN_POS, bn_bits(n));
71  bn_mod(d_1, d_1, n);
72  } while (bn_is_zero(d_1));
73 
74  bn_gcd_ext(t, d_2, NULL, d_1, n);
75  if (bn_sign(d_2) == BN_NEG) {
76  bn_add(d_2, d_2, n);
77  }
78  bn_mul(d_2, d, d_2);
79  bn_mod(d_2, d_2, n);
80  }
81  CATCH_ANY {
82  THROW(ERR_CAUGHT);
83  }
84  FINALLY {
85  bn_free(d);
86  bn_free(n);
87  bn_free(t);
88  }
89 }
