81 void fp2_mul_basic(fp2_t c, fp2_t a, fp2_t b) {
82  dv_t t0, t1, t2, t3, t4;
83 
84  dv_null(t0);
85  dv_null(t1);
86  dv_null(t2);
87  dv_null(t3);
88  dv_null(t4);
89 
90  TRY {
91  dv_new(t0);
92  dv_new(t1);
93  dv_new(t2);
94  dv_new(t3);
95  dv_new(t4);
96 
97  /* Karatsuba algorithm. */
98 
99  /* t2 = a_0 + a_1, t1 = b_0 + b_1. */
100  fp_add(t2, a[0], a[1]);
101  fp_add(t1, b[0], b[1]);
102 
103  /* t3 = (a_0 + a_1) * (b_0 + b_1). */
104  fp_muln_low(t3, t2, t1);
105 
106  /* t0 = a_0 * b_0, t4 = a_1 * b_1. */
107  fp_muln_low(t0, a[0], b[0]);
108  fp_muln_low(t4, a[1], b[1]);
109 
110  /* t2 = (a_0 * b_0) + (a_1 * b_1). */
111  fp_addc_low(t2, t0, t4);
112 
113  /* t1 = (a_0 * b_0) + u^2 * (a_1 * b_1). */
114  fp_subc_low(t1, t0, t4);
115 
116  /* t1 = u^2 * (a_1 * b_1). */
117  for (int i = -1; i > fp_prime_get_qnr(); i--) {
118  fp_subc_low(t1, t1, t4);
119  }
120  for (int i = 0; i <= fp_prime_get_qnr(); i++) {
121  fp_addc_low(t1, t1, t4);
122  }
123 
124  /* c_0 = t1 mod p. */
125  fp_rdc(c[0], t1);
126 
127  /* t4 = t3 - t2. */
128  fp_subc_low(t4, t3, t2);
129 
130  /* c_1 = t4 mod p. */
131  fp_rdc(c[1], t4);
132  }
133  CATCH_ANY {
134  THROW(ERR_CAUGHT);
135  }
136  FINALLY {
137  dv_free(t0);
138  dv_free(t1);
139  dv_free(t2);
140  dv_free(t3);
141  dv_free(t4);
142  }
143 }

145 void fp2_mul_nor_basic(fp2_t c, fp2_t a) {
146  fp2_t t;
147  bn_t b;
148 
149  fp2_null(t);
150  bn_null(b);
151 
152  TRY {
153  fp2_new(t);

155 
156 #ifdef FP_QNRES
157  /* If p = 3 mod 8, (1 + i) is a QNR/CNR. */
158  fp_neg(t[0], a[1]);
159  fp_add(c[1], a[0], a[1]);
160  fp_add(c[0], t[0], a[0]);
161 #else
162  switch (fp_prime_get_mod8()) {
163  case 3:
164  /* If p = 3 mod 8, (1 + u) is a QNR/CNR. */
165  fp_neg(t[0], a[1]);
166  fp_add(c[1], a[0], a[1]);
167  fp_add(c[0], t[0], a[0]);
168  break;
169  case 1:
170  case 5:
171  /* If p = 5 mod 8, (u) is a QNR/CNR. */
172  fp2_mul_art(c, a);
173  break;
174  case 7:
175  /* If p = 7 mod 8, we choose (4 + u) is a QNR/CNR. */
176  fp2_mul_art(t, a);
177  fp2_dbl(c, a);
178  fp2_dbl(c, c);
179  fp2_add(c, c, t);
180  break;
181  default:
182  THROW(ERR_NO_VALID);
183  }
184 #endif
185  }

187  THROW(ERR_CAUGHT);

189  FINALLY {
190  fp2_free(t);

209 void fp2_mul_art(fp2_t c, fp2_t a) {
210  fp_t t;
211 
212  fp_null(t);
213 
214  TRY {
215  fp_new(t);
216 
217 #ifdef FP_QNRES
218  /* (a_0 + a_1 * i) * i = -a_1 + a_0 * i. */
219  fp_copy(t, a[0]);
220  fp_neg(c[0], a[1]);
221  fp_copy(c[1], t);
222 #else
223  /* (a_0 + a_1 * u) * u = (a_1 * u^2) + a_0 * u. */
224  fp_copy(t, a[0]);
225  fp_neg(c[0], a[1]);
226  for (int i = -1; i > fp_prime_get_qnr(); i--) {
227  fp_sub(c[0], c[0], a[1]);
228  }
229  for (int i = 1; i <= fp_prime_get_qnr() + 1; i++) {
230  fp_add(c[0], c[0], a[1]);
231  }
232  fp_copy(c[1], t);
233 #endif
234  }
235  CATCH_ANY {
236  THROW(ERR_CAUGHT);
237  }
238  FINALLY {
239  fp_free(t);
240  }
