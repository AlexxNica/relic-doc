40 void eb_map(eb_t p, const uint8_t *msg, int len) {
41  bn_t k;
42  fb_t t0, t1;
43  int i;
44  uint8_t digest[MD_LEN];
45 
46  bn_null(k);
47  fb_null(t0);
48  fb_null(t1);
49 
50  TRY {
51  bn_new(k);
52  fb_new(t0);
53  fb_new(t1);
54 
55  md_map(digest, msg, len);
56  bn_read_bin(k, digest, MIN(FB_BYTES, MD_LEN));
57  fb_set_dig(p->z, 1);
58 
59  i = 0;
60  while (1) {
61  bn_add_dig(k, k, 1);
62  bn_mod_2b(k, k, FB_BITS);
63  dv_copy(p->x, k->dp, FB_DIGS);
64 
65  eb_rhs(t1, p);
66 
67  /* t0 = 1/x1^2. */
68  fb_sqr(t0, p->x);
69  fb_inv(t0, t0);
70  /* t0 = t1/x1^2. */
71  fb_mul(t0, t0, t1);
72  /* Solve t1^2 + t1 = t0. */
73  if (fb_trc(t0) != 0) {
74  i++;
75  } else {
76  fb_slv(t1, t0);
77  /* x3 = x1, y3 = t1 * x1, z3 = 1. */
78  fb_mul(p->y, t1, p->x);
79  fb_set_dig(p->z, 1);
80 
81  p->norm = 1;
82  break;
83  }
84  }
85  /* Now, multiply by cofactor to get the correct group. */
86  eb_curve_get_cof(k);
87  if (bn_bits(k) < BN_DIGIT) {
88  eb_mul_dig(p, p, k->dp[0]);
89  } else {
90  eb_mul(p, p, k);
91  }
92  }
93  CATCH_ANY {
94  THROW(ERR_CAUGHT);
95  }
96  FINALLY {
97  bn_free(k);
98  fb_free(t0);
99  fb_free(t1);
100  }
101 }
