40 int cp_bls_gen(bn_t d, g2_t q) {

43 
44  bn_null(n);
45 
46  TRY {
47  bn_new(n);
48 
49  g2_get_ord(n);
50 
51  do {
52  bn_rand(d, BN_POS, bn_bits(n));

57  }
58  CATCH_ANY {
